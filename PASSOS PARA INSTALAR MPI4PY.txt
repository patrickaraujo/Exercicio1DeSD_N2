(PASSO 05)
INSTALAR MPI4PY NO CMD
pip install mpi4py


(PASSO 06)
CÓDIGO DE TESTE (teste01.py)

from mpi4py import MPI
comm = MPI.COMM_WORLD
rank = comm.Get_rank()
print ("hello world from process ", rank)

EXECUTANDO O CÓDIGO COM 3 PROCESSOS (-np 3)
mpiexec -np 3 python teste01.py
